【优化第一批：类别权重 + 数据增强】

## 背景
当前数据集存在严重的类别不平衡问题，某些交通方式（如 Airplane）样本数量远少于其他类别。
这会导致模型偏向多数类，对少数类识别效果差。

本批优化通过以下方式解决类别不平衡：
1. 类别权重计算：使用 sqrt_inverse 模式计算类别权重，在损失函数中加权
2. 数据增强：对轨迹段进行随机变换（速度缩放、时间扰动、角度扰动）
3. 过采样：对少数类进行过采样，平衡各类别样本数量

本批修改涉及 common/train_utils.py、common/base_preprocessor.py 和各实验的 train.py。

## 不允许改动的内容
- 模型结构
- 特征提取逻辑
- 缓存机制

## 修改一：train_utils.py 添加类别权重计算函数

在 common/train_utils.py 中添加以下函数：

```python
def compute_class_weights(
    label_encoder,
    all_features_and_labels,
    label_index: int = -1,
    mode: str = 'sqrt_inverse'
) -> torch.Tensor:
    """
    计算类别权重，用于处理类别不平衡问题。

    Args:
        label_encoder: 标签编码器
        all_features_and_labels: 所有特征和标签的列表
        label_index: 标签在 tuple 中的索引
        mode: 权重计算模式
            - 'inverse': 1 / count
            - 'sqrt_inverse': 1 / sqrt(count)
            - 'log_inverse': 1 / log(count + 1)

    Returns:
        class_weights: (num_classes,) 类别权重张量
    """
    from collections import Counter
    import numpy as np

    # 统计每个类别的样本数量
    labels = [item[label_index] for item in all_features_and_labels]
    label_counts = Counter(labels)

    num_classes = len(label_encoder.classes_)
    total_samples = len(labels)

    class_weights = []
    for class_idx, class_name in enumerate(label_encoder.classes_):
        count = label_counts.get(class_name, 0)

        if count == 0:
            weight = 0.0
        elif mode == 'inverse':
            weight = total_samples / (num_classes * count)
        elif mode == 'sqrt_inverse':
            weight = total_samples / (num_classes * np.sqrt(count))
        elif mode == 'log_inverse':
            weight = total_samples / (num_classes * np.log(count + 1))
        else:
            raise ValueError(f"Unknown mode: {mode}")

        class_weights.append(weight)

    class_weights = np.array(class_weights, dtype=np.float32)
    class_weights = class_weights / class_weights.sum() * num_classes

    return torch.from_numpy(class_weights)
```

```python
@staticmethod
def augment_segment(features: np.ndarray, seed: int = None) -> np.ndarray:
    """
    对轨迹段进行数据增强（随机变换）。

    Args:
        features: (N, 9) 特征矩阵
        seed: 随机种子

    Returns:
        augmented_features: (N, 9) 增强后的特征矩阵
    """
    if seed is not None:
        np.random.seed(seed)

    augmented = features.copy()
    N, D = augmented.shape

    # 1. 速度缩放（0.8 ~ 1.2）
    speed_scale = np.random.uniform(0.8, 1.2)
    augmented[:, 2] *= speed_scale  # speed
    augmented[:, 5] *= speed_scale  # distance
    augmented[:, 7] *= speed_scale  # total_distance

    # 2. 时间扰动（0.9 ~ 1.1）
    time_scale = np.random.uniform(0.9, 1.1)
    augmented[:, 6] *= time_scale  # time_diff
    augmented[:, 8] *= time_scale  # total_time

    # 3. 角度扰动（-5° ~ 5°）
    angle_noise = np.random.uniform(-5, 5, size=N)
    augmented[:, 4] += angle_noise  # bearing_change

    return augmented

@staticmethod
def oversample_minority_classes(
    segments: List[Tuple],
    label_encoder,
    label_index: int = -1,
    target_ratio: float = 1.0
) -> List[Tuple]:
    """
    对少数类进行过采样。

    Args:
        segments: 所有轨迹段
        label_encoder: 标签编码器
        label_index: 标签在 tuple 中的索引
        target_ratio: 目标比例（少数类样本数 / 多数类样本数）

    Returns:
        oversampled_segments: 过采样后的轨迹段列表
    """
    from collections import Counter
    import random

    # 统计每个类别的样本数量
    labels = [segment[label_index] for segment in segments]
    label_counts = Counter(labels)

    # 找到多数类的样本数量
    max_count = max(label_counts.values())

    # 对每个少数类进行过采样
    oversampled = list(segments)
    for class_name, count in label_counts.items():
        target_count = int(max_count * target_ratio)
        if count < target_count:
            # 需要过采样
            num_to_add = target_count - count
            class_segments = [s for s in segments if s[label_index] == class_name]

            # 随机采样
            for _ in range(num_to_add):
                segment = random.choice(class_segments)
                oversampled.append(segment)

    return oversampled
```

## 修改一：base_preprocessor.py 添加段级统计特征计算方法

在 BaseGeoLifePreprocessor 类中，找到计算9维基础特征的方法之后，
添加以下新方法：

```python
@staticmethod
def compute_segment_stats(features: np.ndarray) -> np.ndarray:
    """
    计算轨迹段的统计特征（段级，与序列长度无关）。

    基于已计算好的9维特征矩阵，提取能区分交通方式的统计量。

    Args:
        features: (N, 9) 特征矩阵，列顺序：
            [0]latitude [1]longitude [2]speed [3]acceleration
            [4]bearing_change [5]distance [6]time_diff
            [7]total_distance [8]total_time

    Returns:
        stats: (18,) 统计特征向量

    特征说明（共18维）：
        [0]  speed_mean          速度均值
        [1]  speed_std           速度标准差
        [2]  speed_max           速度最大值
        [3]  speed_cv            速度变异系数 = std/mean，区分 Bus/Car 关键特征
        [4]  accel_mean          加速度均值（绝对值）
        [5]  accel_std           加速度标准差
        [6]  accel_max           加速度最大值（绝对值）
        [7]  bearing_change_mean 方向变化均值，区分直线/曲线行驶
        [8]  bearing_change_std  方向变化标准差
        [9]  stop_ratio          停止比例（速度<0.5m/s 的点占比），Bus特征
        [10] high_speed_ratio    高速比例（速度>15m/s 的点占比），Train/Airplane特征
        [11] linearity           直线度 = 直线距离/累计距离，Walk特征
        [12] total_distance      总行驶距离
        [13] total_time          总行驶时间
        [14] avg_segment_speed   平均段速度 = 总距离/总时间
        [15] speed_entropy       速度分布熵，衡量速度变化规律性
        [16] accel_sign_changes  加速度符号变化次数（归一化），Bus走走停停特征
        [17] max_sustained_speed 最大持续高速时长比例，区分 Subway/Train
    """
    eps = 1e-8
    N = len(features)

    if N == 0:
        return np.zeros(18, dtype=np.float32)

    speed = features[:, 2].clip(0)           # 速度非负
    accel = np.abs(features[:, 3])           # 加速度取绝对值
    bearing = np.abs(features[:, 4])         # 方向变化取绝对值
    distance = features[:, 5].clip(0)
    time_diff = features[:, 6].clip(0)
    total_dist = features[-1, 7] if features[-1, 7] > 0 else distance.sum()
    total_time = features[-1, 8] if features[-1, 8] > 0 else time_diff.sum()

    # [0-2] 速度统计
    speed_mean = float(np.mean(speed))
    speed_std  = float(np.std(speed))
    speed_max  = float(np.max(speed))

    # [3] 速度变异系数（Bus走走停停 → CV高，Car匀速 → CV低）
    speed_cv = speed_std / (speed_mean + eps)

    # [4-6] 加速度统计
    accel_mean = float(np.mean(accel))
    accel_std  = float(np.std(accel))
    accel_max  = float(np.max(accel))

    # [7-8] 方向变化统计
    bearing_mean = float(np.mean(bearing))
    bearing_std  = float(np.std(bearing))

    # [9] 停止比例（速度 < 0.5 m/s）
    stop_ratio = float(np.mean(speed < 0.5))

    # [10] 高速比例（速度 > 15 m/s ≈ 54 km/h）
    high_speed_ratio = float(np.mean(speed > 15.0))

    # [11] 直线度（末点直线距离 / 累计行驶距离）
    if total_dist > eps:
        lat_start, lon_start = features[0, 0], features[0, 1]
        lat_end,   lon_end   = features[-1, 0], features[-1, 1]
        # 简单欧氏近似（小范围内误差可忽略）
        straight_dist = np.sqrt(
            ((lat_end - lat_start) * 111300) ** 2 +
            ((lon_end - lon_start) * 111300 * np.cos(np.radians(lat_start))) ** 2
        )
        linearity = float(min(straight_dist / (total_dist + eps), 1.0))
    else:
        linearity = 0.0

    # [12-14] 总距离、总时间、平均速度
    total_distance    = float(total_dist)
    total_time_val    = float(total_time)
    avg_segment_speed = float(total_dist / (total_time + eps))

    # [15] 速度分布熵（将速度分成10个区间计算熵）
    if speed_max > eps:
        hist, _ = np.histogram(speed, bins=10, range=(0, speed_max + eps))
        hist = hist / (hist.sum() + eps)
        hist = hist[hist > 0]
        speed_entropy = float(-np.sum(hist * np.log(hist + eps)))
    else:
        speed_entropy = 0.0

    # [16] 加速度符号变化次数（归一化到[0,1]）
    if N > 1:
        raw_accel = features[:, 3]
        sign_changes = np.sum(np.diff(np.sign(raw_accel)) != 0)
        accel_sign_changes = float(sign_changes / (N - 1))
    else:
        accel_sign_changes = 0.0

    # [17] 最大持续高速时长比例（连续速度>10m/s的最长片段）
    high_speed_mask = speed > 10.0
    max_sustained = 0
    current_run = 0
    for v in high_speed_mask:
        if v:
            current_run += 1
            max_sustained = max(max_sustained, current_run)
        else:
            current_run = 0
    max_sustained_speed = float(max_sustained / N)

    stats = np.array([
        speed_mean, speed_std, speed_max, speed_cv,
        accel_mean, accel_std, accel_max,
        bearing_mean, bearing_std,
        stop_ratio, high_speed_ratio,
        linearity,
        total_distance, total_time_val, avg_segment_speed,
        speed_entropy,
        accel_sign_changes,
        max_sustained_speed
    ], dtype=np.float32)

    # 处理 NaN/Inf
    stats = np.nan_to_num(stats, nan=0.0, posinf=0.0, neginf=0.0)

    return stats
```

## 修改二：base_model.py 新增 HierarchicalTransportationClassifier 子类

在 BaseTransportationClassifier 类定义之后，追加新类：

```python
class HierarchicalTransportationClassifier(BaseTransportationClassifier):
    """
    层次化时序编码器（论文创新点一）

    核心思想：
        局部编码器捕捉短时运动模式（如加速、转弯），
        全局编码器捕捉运动模式的变化趋势（如 Bus 的周期性停靠）。

    架构：
        输入序列 (B, T, D)
        → 分段为 num_segments 个子序列 (B, num_segments, seg_len, D)
        → 局部 Bi-LSTM 编码每段 (B, num_segments, local_hidden*2)
        → 全局 Bi-LSTM 编码段序列 (B, global_hidden*2)
        → Attention Pooling
        → 分类器
    """

    def __init__(
        self,
        input_dims: list,
        hidden_dims: list,
        num_layers: int = 2,
        num_classes: int = 7,
        dropout: float = 0.3,
        num_segments: int = 5,       # 将序列分成几段
        local_hidden: int = 64,      # 局部编码器隐藏维度
        global_hidden: int = 128,    # 全局编码器隐藏维度
    ):
        super().__init__(
            input_dims=input_dims,
            hidden_dims=hidden_dims,
            num_layers=num_layers,
            num_classes=num_classes,
            dropout=dropout,
        )

        self.num_segments = num_segments
        self.local_hidden = local_hidden
        self.global_hidden = global_hidden

        # 为每个输入模态建立局部编码器
        self.local_encoders = nn.ModuleList([
            nn.LSTM(
                input_size=input_dim,
                hidden_size=local_hidden,
                num_layers=1,
                batch_first=True,
                bidirectional=True
            )
            for input_dim in input_dims
        ])

        # 全局编码器：输入是所有模态局部表示的拼接
        global_input_dim = local_hidden * 2 * len(input_dims)
        self.global_encoder = nn.LSTM(
            input_size=global_input_dim,
            hidden_size=global_hidden,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0,
            bidirectional=True
        )

        # 全局 Attention
        self.global_attn = nn.Linear(global_hidden * 2, 1)

        # 分类器（替换父类的 classifier）
        self.classifier = nn.Sequential(
            nn.Linear(global_hidden * 2, global_hidden),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(global_hidden, num_classes)
        )

    def forward(self, *inputs: torch.Tensor) -> torch.Tensor:
        """
        Args:
            inputs: 每个元素形状 (batch, seq_len, input_dim_i)

        Returns:
            logits: (batch, num_classes)
        """
        batch_size = inputs[0].size(0)
        seq_len    = inputs[0].size(1)

        # 将序列分段
        seg_len = seq_len // self.num_segments
        if seg_len == 0:
            seg_len = 1

        # 局部编码：对每个模态的每个子段编码
        local_reprs = []  # 每个模态的局部表示序列
        for i, (inp, local_enc) in enumerate(zip(inputs, self.local_encoders)):
            # inp: (batch, seq_len, dim)
            seg_reprs = []
            for s in range(self.num_segments):
                start = s * seg_len
                end   = start + seg_len if s < self.num_segments - 1 else seq_len
                seg   = inp[:, start:end, :]  # (batch, seg_len, dim)
                out, _ = local_enc(seg)
                # 取每段的 Attention 加权表示
                attn_w = torch.softmax(
                    out.new_zeros(out.size(0), out.size(1), 1).fill_(1.0 / out.size(1)),
                    dim=1
                )
                # 简化版：取最后时间步，后续可换成 attention
                seg_repr = out[:, -1, :]  # (batch, local_hidden*2)
                seg_reprs.append(seg_repr.unsqueeze(1))  # (batch, 1, local_hidden*2)

            modal_repr = torch.cat(seg_reprs, dim=1)  # (batch, num_segments, local_hidden*2)
            local_reprs.append(modal_repr)

        # 拼接所有模态的局部表示
        combined = torch.cat(local_reprs, dim=2)
        # (batch, num_segments, local_hidden*2 * num_modalities)

        # 全局编码
        global_out, _ = self.global_encoder(combined)
        # (batch, num_segments, global_hidden*2)

        # 全局 Attention Pooling
        attn_weights = torch.softmax(self.global_attn(global_out), dim=1)
        global_repr  = (attn_weights * global_out).sum(dim=1)
        # (batch, global_hidden*2)

        # 分类
        logits = self.classifier(global_repr)
        return logits

    def predict_proba(self, *inputs: torch.Tensor):
        """预测类别和概率"""
        self.eval()
        with torch.no_grad():
            logits = self.forward(*inputs)
            probs  = torch.softmax(logits, dim=1)
            preds  = torch.argmax(probs, dim=1)
        return preds, probs
```

# 第四批：exp5 对比学习重做（论文核心创新）

【优化第四批：exp5 跨模态对比学习】

## 背景
当前 exp5 的弱监督用 MSE loss 直接拉近轨迹表示和上下文表示，
理论上不成立（两个表示的语义空间不同，不应该强迫它们数值相等）。

改为跨模态对比学习（Cross-modal Contrastive Alignment）：
- 同一轨迹段的轨迹特征和上下文特征（空间+天气）互为正样本
- 不同轨迹段的互为负样本
- 用 InfoNCE loss（NT-Xent）对齐两个模态的表示空间

这是论文中"Cross-modal Contrastive Alignment for Trajectory Understanding"
的核心贡献，与现有 GTA-Seg 类工作有明显区别。

## 完整替换 WeaklySupervisedContextModel 类

将 exp5/src/model_weak_supervision.py 中的
WeaklySupervisedContextModel 类完整替换为以下实现：

```python
class WeaklySupervisedContextModel(nn.Module):
    """
    跨模态对比学习模型 (Exp5 - 论文核心创新)

    与 Exp4 的关键区别：
        Exp4: traj + spatial + weather 硬拼接 → 分类器
        Exp5: traj → 分类器（主路径）
              spatial + weather → 上下文编码器（辅助路径）
              两路径用 InfoNCE loss 在表示空间对齐

    理论依据：
        同一轨迹段的轨迹特征和上下文特征描述同一物理事件，
        在表示空间中应该比其他轨迹段的表示更相似（对比学习假设）。
        这种软约束比 MSE 更合理，因为不要求两个表示数值相等，
        只要求相对距离关系正确。

    论文中的名称：
        Cross-modal Contrastive Alignment (CCA) loss
    """

    def __init__(self,
                 trajectory_feature_dim: int = 9,
                 spatial_feature_dim: int = 15,
                 weather_feature_dim: int = 12,
                 hidden_dim: int = 128,
                 num_layers: int = 2,
                 num_classes: int = 7,
                 dropout: float = 0.3,
                 context_loss_type: str = 'infonce',  # 改为 infonce
                 context_loss_weight: float = 0.05,
                 temperature: float = 0.07):          # InfoNCE 温度参数
        super().__init__()

        self.trajectory_feature_dim = trajectory_feature_dim
        self.spatial_feature_dim    = spatial_feature_dim
        self.weather_feature_dim    = weather_feature_dim
        self.hidden_dim             = hidden_dim
        self.num_classes            = num_classes
        self.context_loss_type      = context_loss_type
        self.context_loss_weight    = context_loss_weight
        self.temperature            = temperature

        # ===== 轨迹编码器（主路径）=====
        self.trajectory_lstm = nn.LSTM(
            input_size=trajectory_feature_dim,
            hidden_size=hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0,
            bidirectional=True
        )
        # Attention Pooling（替代取最后时间步）
        self.traj_attn = nn.Linear(hidden_dim * 2, 1)

        # 轨迹投影头（用于对比学习，不影响分类）
        self.traj_proj = nn.Sequential(
            nn.Linear(hidden_dim * 2, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim // 2)
        )

        # ===== 上下文编码器（辅助路径）=====
        self.spatial_lstm = nn.LSTM(
            input_size=spatial_feature_dim,
            hidden_size=hidden_dim // 2,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0,
            bidirectional=True
        )
        self.spatial_attn = nn.Linear(hidden_dim, 1)

        self.weather_lstm = nn.LSTM(
            input_size=weather_feature_dim,
            hidden_size=hidden_dim // 4,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0,
            bidirectional=True
        )
        self.weather_attn = nn.Linear(hidden_dim // 2, 1)

        # 上下文融合
        context_dim = hidden_dim + hidden_dim // 2  # spatial + weather
        self.context_fusion = nn.Sequential(
            nn.Linear(context_dim, hidden_dim),
            nn.ReLU(),
            nn.Dropout(dropout)
        )
        # 上下文投影头（与轨迹投影头对齐到同一空间）
        self.context_proj = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim // 2)
        )

        # ===== 分类器（仅使用轨迹表示）=====
        self.classifier = nn.Sequential(
            nn.Linear(hidden_dim * 2, hidden_dim),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_dim, num_classes)
        )

    def _attention_pool(self, lstm_out: torch.Tensor,
                        attn_layer: nn.Linear) -> torch.Tensor:
        """对 LSTM 输出做 Attention Pooling"""
        attn_weights = torch.softmax(attn_layer(lstm_out), dim=1)  # (B, T, 1)
        return (attn_weights * lstm_out).sum(dim=1)                 # (B, hidden)

    def forward(self, trajectory_features: torch.Tensor,
                spatial_features: torch.Tensor,
                weather_features: torch.Tensor,
                return_context: bool = False):
        """
        Args:
            trajectory_features: (B, T, 9)
            spatial_features:    (B, T, 15)
            weather_features:    (B, T, 12)
            return_context:      是否返回投影向量（训练时用于计算 CCA loss）

        Returns:
            若 return_context=False: logits (B, num_classes)
            若 return_context=True:  (logits, traj_proj, context_proj)
        """
        # 轨迹编码
        traj_out, _   = self.trajectory_lstm(trajectory_features)
        traj_repr     = self._attention_pool(traj_out, self.traj_attn)  # (B, hidden*2)

        # 上下文编码
        spatial_out, _ = self.spatial_lstm(spatial_features)
        spatial_repr   = self._attention_pool(spatial_out, self.spatial_attn)  # (B, hidden)

        weather_out, _ = self.weather_lstm(weather_features)
        weather_repr   = self._attention_pool(weather_out, self.weather_attn)  # (B, hidden//2)

        context_combined = torch.cat([spatial_repr, weather_repr], dim=1)
        context_repr     = self.context_fusion(context_combined)        # (B, hidden)

        # 分类（仅使用轨迹表示）
        logits = self.classifier(traj_repr)

        if return_context:
            # 投影到对比学习空间（L2 归一化）
            traj_z    = F.normalize(self.traj_proj(traj_repr), dim=1)      # (B, hidden//2)
            context_z = F.normalize(self.context_proj(context_repr), dim=1) # (B, hidden//2)
            return logits, traj_z, context_z
        else:
            return logits

    def compute_context_loss(self, traj_z: torch.Tensor,
                             context_z: torch.Tensor) -> torch.Tensor:
        """
        InfoNCE（NT-Xent）跨模态对比损失

        对于 batch 中的每个样本 i：
        - 正样本对：(traj_z[i], context_z[i])，同一轨迹段的两个模态表示
        - 负样本对：(traj_z[i], context_z[j]) for j≠i，不同轨迹段

        损失公式：
            L = -1/N * sum_i log(
                exp(sim(traj_z[i], context_z[i]) / τ) /
                sum_j exp(sim(traj_z[i], context_z[j]) / τ)
            )

        Args:
            traj_z:    (B, D) L2归一化的轨迹投影
            context_z: (B, D) L2归一化的上下文投影

        Returns:
            loss: 标量
        """
        B = traj_z.size(0)
        if B < 2:
            # batch 太小无法做对比学习，退化为 MSE
            return F.mse_loss(traj_z, context_z)

        # 相似度矩阵 (B, B)：traj_z[i] 与所有 context_z[j] 的余弦相似度
        # 由于已 L2 归一化，点积即余弦相似度
        sim_matrix = torch.matmul(traj_z, context_z.T) / self.temperature  # (B, B)

        # 正样本在对角线上
        labels = torch.arange(B, device=traj_z.device)

        # 双向对比：traj→context 和 context→traj
        loss_t2c = F.cross_entropy(sim_matrix, labels)
        loss_c2t = F.cross_entropy(sim_matrix.T, labels)

        return (loss_t2c + loss_c2t) / 2.0

    def predict(self, trajectory_features, spatial_features, weather_features):
        """预测类别和概率"""
        self.eval()
        with torch.no_grad():
            logits = self.forward(trajectory_features, spatial_features,
                                  weather_features, return_context=False)
            probs = F.softmax(logits, dim=1)
            preds = torch.argmax(probs, dim=1)
        return preds, probs
```
